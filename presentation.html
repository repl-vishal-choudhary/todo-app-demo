Review Classification Logic
// Hierarchical issue classification system enum ReviewAction { IMMEDIATE_REJECTION = "block_merge", MANDATORY_FIX = "require_changes", QUALITY_IMPROVEMENT = "suggest_fix" } function classifyIssue(violation) { if (isSecurityRisk(violation) || isBreakingChange(violation)) { return ReviewAction.IMMEDIATE_REJECTION; } if (isTypeViolation(violation) || isAccessibilityFail(violation)) { return ReviewAction.MANDATORY_FIX; } return ReviewAction.QUALITY_IMPROVEMENT; }
